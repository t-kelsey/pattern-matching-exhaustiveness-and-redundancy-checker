=== data types ===
data Unit where
  tt : Unit

data Nat where
  zero : Nat
  succ : Nat -> Nat

data List where
  nil  : List
  cons : Nat -> List -> List

data OneOfThose where
  nat  : Nat -> Nat -> List -> OneOfThose
  list : List -> OneOfThose
  unit : Unit -> OneOfThose

=== pattern matrix ===
(nat x y z) w zero
(list x) w z
(unit x) (succ (succ y)) z
(unit x) y z
