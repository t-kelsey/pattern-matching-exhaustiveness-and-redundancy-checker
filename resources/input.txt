=== data types ===

data Unit where
  tt : Unit

data Nat where
  zero : Nat
  suc  : Nat -> Nat

data List where
  nil  : List
  cons : Nat -> List -> List

data OneOfThose where
  nat  : Nat -> Nat -> List -> OneOfThose
  list : List -> OneOfThose
  unit : Unit -> OneOfThose
