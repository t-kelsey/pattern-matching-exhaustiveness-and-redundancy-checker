-- Testing the parser for pattern matrices
-- Case: Rows mix or-patterns and constructor patterns
-- Expected: ((nat x zero y) | (nat zero zero nil)) x
--           (list (cons zero nil)) x
--           (unit x) x

=== data types ===
data Unit where
  tt : Unit

data Nat where
  zero : Nat
  succ : Nat -> Nat

data List where
  nil  : List
  cons : Nat -> List -> List

data OneOfThose where
  nat  : Nat -> Nat -> List -> OneOfThose
  list : List -> OneOfThose
  unit : Unit -> OneOfThose

=== pattern matrix ===
(nat x zero y) | (nat zero zero nil)  x
(list (cons zero nil))          x
(unit x)                  x
